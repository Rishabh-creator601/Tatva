
# objective 
To determine the export by the panel 


folder name  : "Tatva Exports"
location : desktop 

# IMAGES 
Images can be exported separately by a carousel or the whole page images. 
The images will be exported. under the {page name }/{caraousel name}/images 


# TEXT
the text will be exported under the {page name }/{caraousel name}

all will be in one file Separately, by the carousel 

in the format 

<----NOTE 1 --- >

note content 
note message : {special message}


<---- NOTE 2 --->
note content 
note message : {special message }



# Links 
Exported links can be of the Categorized  or uncategorized. 
the text will be exported under the {page name }/{caraousel name}/{category name if avail }

in the format 


title : {title}
link : {link}
message :{if available else do not even mention this }
